      gg.num[gg.s==uus[i]]<-tts[i]
    }

    if(any(nn>2))
    {
      gg.num[nn<=2]<-as.numeric(gg.s[nn<=2])
      uus<-unique(gg.s[nn>2])
      tts<-seq((max(gg.num,na.rm=T)+1),(max(gg.num,na.rm=T)+length(uus)))
      for(i in 1:length(uus))
      {
        gg.num[gg.s==uus[i]]<-tts[i]
      }

      gg.num<-as.numeric(gg.s)
    }
  }
  ss.test$group<-gg.num
  LM_procg<-rbind(LM_procg,ss.test)
  #cat(rr,"\n")
}

#B punishment side?

Here we use the temperature data to identify problem chambers. For thermal tolerance, if the average temperature over the assay is less than 35, we flagged it as problematic. For learning and memory, if the absolute value of the average temperature was greater than 5, we flagged it as problematic. There are very few chambers flagged as problematic for learning that are not also problematic for thermal tolerance (17 records). There are more that are flagged for thermal tolerance but not for learning (7048 records). This could be occuring when the punishment side is NOT the side with the problem.
